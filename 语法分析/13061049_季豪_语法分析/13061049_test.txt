const a=3,b=4,c=5;
var x1x,s,a1,b1,c1;
procedure Circle;
  begin
    if a1=1 then
      begin
        s:=1;
        write(s)
      end
    else
      begin
        a1:=a-1;
        call Circle
      end
  end;

begin
  a1:=a;
  b1:=b;
  c1:=c;
  ;
  read(x1x);
  write(x1x);
  if x1x<2 then
    begin
      b1:=2;
      c1:=3;
      call Circle
    end
  else
    begin
      s:=2;
      write(s);
      while x1x>=2 do
        x1x:=x1x-1
    end;
  write(a1,b1,c1)
end.
